#include<stdio.h>
#include<stdlib.h>

typedef struct No{
    int valor;
    struct No *esquerda;
    struct No *direita;
}No;

No* criarNo(int valor){
    No* novo = (No*) malloc (sizeof(No));
    novo->valor = valor;
    novo->esquerda = NULL;
    novo->direita = NULL;
    return novo;
}

No* inserir(No* raiz, int valor)
{
    if (raiz == NULL)
    {
        return criarNo(valor);
    }
    
    if (valor < raiz->valor)  
    {
        raiz->esquerda = inserir(raiz->esquerda, valor);
    }
    else if (valor > raiz->valor)
    {
        raiz->direita = inserir(raiz->direita, valor);
    }
   
    return raiz;
}

int busca (No* raiz, int valor)
{
    if (raiz == NULL)
    {
        return 0;   
    }
    
    if(raiz->valor == valor)
    {
        return 1;
    }
    
    if(valor < raiz->valor)
    {
        return busca(raiz->esquerda, valor);  
    }
    else 
    {
        return busca(raiz->direita, valor);
    }
}
 
void exibirEmOrdem(No* raiz)
{
    if(raiz != NULL){
        exibirEmOrdem(raiz->esquerda);
        printf("%d ", raiz->valor);  
        exibirEmOrdem(raiz->direita);
    }
}

void imprimirArvora(No* raiz, int espaco)  
{
    if(raiz == NULL)
        return;
    
    espaco += 5;
    imprimirArvora(raiz->direita, espaco);  
    
    printf("\n");
    for(int i = 5; i < espaco; i++) 
    {
        printf(" "); 
    }
    printf("%d\n", raiz->valor);  
    imprimirArvora(raiz->esquerda, espaco);  
}

void liberarArvore(No* raiz) 
{
    if(raiz != NULL)
    {
        liberarArvore(raiz->esquerda);  
        liberarArvore(raiz->direita);   
        free(raiz);  
    }
}

int main(){
    No* raiz = NULL;  
    int opcao;
    int valor;
    
    do
    {
        printf("\n=== Menu ===\n"); 
        printf("1 - Inserir Elemento\n");
        printf("2 - Buscar Elemento\n");
        printf("3 - Exibir em Ordem\n");
        printf("4 - Exibir Árvore\n");
        printf("5 - Reiniciar Árvore\n");
        printf("0 - Sair\n");
        
        printf("Escolha: ");
        scanf("%d", &opcao);
        
        switch(opcao){
            
            case 1:  
                printf("Digite o valor: ");
                scanf("%d", &valor);
                raiz = inserir(raiz, valor);
                printf("Valor inserido!\n");
                break;
                
            case 2:  
                printf("Digite o valor: ");
                scanf("%d", &valor);
                if(busca(raiz, valor)){
                    printf("Valor Encontrado!\n");
                }
                else
                {
                    printf("Valor Não Encontrado\n");
                }
                break;
                
            case 3:  
                printf("Impressão em ordem: ");
                exibirEmOrdem(raiz);
                printf("\n");
                break;
                
            case 4:  
                printf("Impressão em Arvore: \n");
                imprimirArvora(raiz, 0); 
                printf("\n");
                break;
                
            case 5: 
                liberarArvore(raiz);
                raiz = NULL;
                printf("Árvore reiniciada!\n");
                break;
                
            case 0: 
                printf("Saindo...\n");
                break;
